$$
\begin{gathered}
\large S_x 
    = x_1 + x_2 + ... \; x_n
    = \small \sum_{i=1}^{n} x_i \\
\large S_y 
    = y_1 + y_2 + ... \; y_n
    = \small \sum_{i=1}^{n} y_i \\

\\

\large S_{x^2}
    = x_1^2 + x_2^2 + ... \; x_n^2
    = \small \sum_{i=1}^{n} x_i^2 \\
\large S_{y^2}
    = y_1^2 + y_2^2 + ... \; y_n^2
    = \small \sum_{i=1}^{n} y_i^2 \\

\\

\large \mu_x 
    = \frac{S_x}{n}
    = \frac{\small \sum_{i=1}^{n} x_i}{n} \\
\large \mu_y 
    = \frac{S_y}{n}
    = \frac{\small \sum_{i=1}^{n} y_i}{n} \\

\\

\large S_{xx}
    = (x_1-\mu_x)^2 + (x_2-\mu_x)^2 + ... \; (x_n-\mu_x)^2
    = \small \sum_{i=1}^{n} (x_i-\mu_x)^2 \\
\large S_{yy}
    = (y_1-\mu_y)^2 + (y_2-\mu_y)^2 + ... \; (y_n-\mu_y)^2
    = \small \sum_{i=1}^{n} (y_i-\mu_y)^2 \\
\large S_{xy} = 
    (x_1-\mu_x)(y_1-\mu_y) + 
    (x_2-\mu_x)(y_2-\mu_y) + ... \; 
    (x_n-\mu_x)(y_n-\mu_y)
    = \small \sum_{i=1}^{n} (x_i-\mu_x)(y_i-\mu_y) \\

\\

\large \sigma_x 
    = \sqrt{\frac{S_{xx}}{n}}
    = \sqrt{\frac{S_{x^2}-n{\mu_x}^2}{n}} \\
\large \sigma_y
    = \sqrt{\frac{S_{yy}}{n}}
    = \sqrt{\frac{S_{y^2}-n{\mu_y}^2}{n}} \\

\\

\large S_{x^2} = n(\sigma_x^2 + \mu_x^2) \\
\large S_{y^2} = n(\sigma_y^2 + \mu_y^2) \\

\\

\large X_i = \frac{x_i-\mu_x}{\sigma_x} \\
\large Y_i = \frac{y_i-\mu_y}{\sigma_y} \\

\\

\large r 
    \large = \frac{{X_1}{Y_1} + {X_2}{Y_2} + ... \; {X_n}{Y_n}}{n}
    = \small \frac{\sum_{i=1}^{n} {X_i}{Y_i}}{n} \\
\large r 
    \large = \frac{({x_1}{y_1} + {x_2}{y_2} + ... \; {x_n}{x_n})-n\mu_x\mu_y}{n}
    = \small \frac{(\sum_{i=1}^{n} {x_i}{y_i})-n\mu_x\mu_y}{n} \\

\\

\large r 
    = \frac{S_{xy}}{n\sigma_x\sigma_y}
    = \frac{S_{xy}}{\sqrt{S_{xx}}\sqrt{S_{yy}}}
\end{gathered}
$$
